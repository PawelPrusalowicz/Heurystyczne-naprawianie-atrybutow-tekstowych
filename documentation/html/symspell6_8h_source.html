126  int levenshtein_dist(char const* word1, char const* word2) {
134  int size1 = strlen(word1) + 1, size2 = strlen(word2) + 1;
135  int suppr_dist, insert_dist, subs_dist;
136  int* dist = new int[(size1)*size2];
137 
138  for (int i = 0; i < size1; ++i)
139  dist[size2*i] = i;
140  for (int j = 0; j < size2; ++j)
141  dist[j] = j;
142  for (int i = 1; i < size1; ++i) {
143  for (int j = 1; j < size2; ++j) {
144  suppr_dist = dist[size2*(i - 1) + j] + 1;
145  insert_dist = dist[size2*i + j - 1] + 1;
146  subs_dist = dist[size2*(i - 1) + j - 1];
147  if (word1[i - 1] != word2[j - 1]) { // word indexes are implemented differently.
148  subs_dist += 1;
149  }
150  dist[size2*i + j] = mini(suppr_dist, insert_dist, subs_dist);
151  }
152  }
153 
154  // --------------------------------------------------------
155  int res = dist[size1 * size2 - 1];
156  delete[] dist;
157  return(res);
158  }

166  int size1 = strlen(word1) + 1, size2 = strlen(word2) + 1;
167  int suppr_dist, insert_dist, subs_dist, val;
168  int* dist = new int[size1*size2];
169 
170  for (int i = 0; i < size1; ++i)
171  dist[size2*i] = i;
172  for (int j = 0; j < size2; ++j)
173  dist[j] = j;
174  for (int i = 1; i < size1; ++i) {
175  for (int j = 1; j < size2; ++j) {
176  suppr_dist = dist[size2*(i - 1) + j] + 1;
177  insert_dist = dist[size2*i + j - 1] + 1;
178  subs_dist = dist[size2*(i - 1) + j - 1];
179  if (word1[i - 1] != word2[j - 1]) // word indexes are implemented differently.
180  subs_dist += 1;
181  val = mini(suppr_dist, insert_dist, subs_dist);
182  if (((i >= 2) && (j >= 2)) && ((word1[i - 1] == word2[j - 2]) && (word1[i - 2] == word2[j - 1])))
183  val = min(dist[size2*(i - 2) + j - 2] + 1, val);
184  dist[size2*i + j] = val;
185  }
186  }
187

584  bool CreateDictionaryEntry(const char * key, int64_t count, SuggestionStage * staging = nullptr)
585  {
586  int keyLen = strlen(key);
587  if (count <= 0)
588  {
589  if (this->countThreshold > 0) return false; // no point doing anything if count is zero, as it can't change anything
590  count = 0;
591  }
592 
593  int64_t countPrevious = -1;
594  auto belowThresholdWordsFinded = belowThresholdWords.find(key);
595  auto wordsFinded = words.find(key);
596 
597  // look first in below threshold words, update count, and allow promotion to correct spelling word if count reaches threshold
598  // threshold must be >1 for there to be the possibility of low threshold words
599  if (countThreshold > 1 && belowThresholdWordsFinded != belowThresholdWordsEnd)
600  {
601  countPrevious = belowThresholdWordsFinded->second;
602  // calculate new count for below threshold word
603  count = ((std::numeric_limits<int64_t>::max)() - countPrevious > count) ? countPrevious + count : (std::numeric_limits<int64_t>::max)();
604  // has reached threshold - remove from below threshold collection (it will be added to correct words below)
605  if (count >= countThreshold)
606  {
607  belowThresholdWords.erase(key);
608  belowThresholdWordsEnd = belowThresholdWords.end();
609  }
610  else
611  {
612  belowThresholdWords[key] = count;
613  belowThresholdWordsEnd = belowThresholdWords.end();
614  return false;
615  }
616  }
617  else if (wordsFinded != wordsEnd)
618  {
619  countPrevious = wordsFinded->second;
620  count = ((std::numeric_limits<int64_t>::max)() - countPrevious > count) ? countPrevious + count : (std::numeric_limits<int64_t>::max)();
621  words[key] = count;
622  return false;
623  }
624  else if (count < CountThreshold())
625  {
626  belowThresholdWords[key] = count;
627  belowThresholdWordsEnd = belowThresholdWords.end();
628  return false;
629  }
630 
631  words[key] = count;
632 
633  if (keyLen > maxDictionaryWordLength)
634  maxDictionaryWordLength = keyLen;
635 
636  EditsPrefix(key, edits);
637 
638  if (staging != nullptr)
639  {
640  auto editsEnd = edits.end();
641  for (auto it = edits.begin(); it != editsEnd; ++it)
642  {
643  staging->Add(*it, _strdup(key));
644  }
645  }
646  else
647  {
648  auto editsEnd = edits.end();
649  for (auto it = edits.begin(); it != editsEnd; ++it)
650  {
651  size_t deleteHash = *it;
652  auto deletesFinded = deletes.find(deleteHash);
653  if (deletesFinded != deletesEnd)
654  {
655  char* tmp = new char[keyLen + 1];
656  std::memcpy(tmp, key, keyLen);
657  tmp[keyLen] = '\0';
658 
659  //delete[] deletes[deleteHash][deletesFinded->second.size() - 1];

661  deletesEnd = deletes.end();
662  }
663  else
664  {
665  char* tmp = new char[keyLen + 1];
666  std::memcpy(tmp, key, keyLen);
667  tmp[keyLen] = '\0';

753  void Lookup(const char * input, Verbosity verbosity, int32_t maxEditDistance, bool includeUnknown, vector<std::unique_ptr<symspell::SuggestItem>> & suggestions)
754  {
755  mtx.lock();
756  suggestions.clear();
757  edits.clear();
758  candidates.reserve(32);
759 
760  //verbosity=Top: the suggestion with the highest term frequency of the suggestions of smallest edit distance found
761  //verbosity=Closest: all suggestions of smallest edit distance found, the suggestions are ordered by term frequency
762  //verbosity=All: all suggestions <= maxEditDistance, the suggestions are ordered by edit distance, then by term frequency (slower, no early termination)
763 
764  // maxEditDistance used in Lookup can't be bigger than the maxDictionaryEditDistance
765  // used to construct the underlying dictionary structure.
766  if (maxEditDistance > MaxDictionaryEditDistance()) throw std::invalid_argument("maxEditDistance");
767  int64_t suggestionCount = 0;
768  size_t suggestionsLen = 0;
769  auto wordsFinded = words.find(input);
770  int inputLen = strlen(input);
771  // early exit - word is too big to possibly match any words
772  if (inputLen - maxEditDistance > maxDictionaryWordLength)
773  {
774  if (includeUnknown && (suggestionsLen == 0))
775  {
776  std::unique_ptr<SuggestItem> unq(new SuggestItem(_strdup(input), maxEditDistance + 1, 0));
777  suggestions.push_back(std::move(unq));
778  }
779 
780  mtx.unlock();
781  return;
782  }
783 
784  // quick look for exact match
785 
786  if (wordsFinded != wordsEnd)
787  {
788  suggestionCount = wordsFinded->second;
789 
790  {
791  std::unique_ptr<SuggestItem> unq(new SuggestItem(_strdup(input), 0, suggestionCount));
792  suggestions.push_back(std::move(unq));
793  }
794 
795  ++suggestionsLen;
796  // early exit - return exact match, unless caller wants all matches

1091  shared_ptr<WordSegmentationItem> WordSegmentation(const char* input, size_t maxEditDistance, size_t maxSegmentationWordLength)
1092  {
1093  size_t inputLen = strlen(input);
1094  int arraySize = min(maxSegmentationWordLength, strlen(input));
1095  std::vector<shared_ptr<WordSegmentationItem>> compositions;
1096  compositions.resize(arraySize);
1097 
1098  for (size_t i = 0; i < arraySize; ++i)
1099  {
1100  std::shared_ptr<WordSegmentationItem> unq(new WordSegmentationItem());
1101  compositions[i] = std::move(unq);
1102  }
1103 
1104  int circularIndex = -1;
1105 
1106  for (int j = 0; j < inputLen; ++j)
1107  {
1108  //inner loop (row): all possible part lengths (from start position): part can't be bigger than longest word in dictionary (other than long unknown word)
1109  int imax = min(inputLen - j, maxSegmentationWordLength);
1110  for (int i = 1; i <= imax; ++i)
1111  {
1112  char* part = new char[i + 1];
1113  std::memcpy(part, input + j, i);
1114  part[i] = '\0';
1115 
1116  int separatorLength = 0;
1117  int topEd = 0;
1118  double topProbabilityLog = 0;
1119  char* topResult = nullptr;
1120 
1121  if (isspace(part[0]))
1122  {
1123  size_t partLen = strlen(part);
1124  char* tmp = new char[partLen];
1125  std::memcpy(tmp, part + 1, partLen - 1);
1126  tmp[(i - j)] = '\0';
1127 
1128  delete[] part;
1129  part = tmp;
1130  }
1131  else
1132  {
1133  //add ed+1: space did not exist, had to be inserted
1134  separatorLength = 1;
1135  }
1136 
1137  //remove space from part1, add number of removed spaces to topEd
1138  topEd += strlen(part);
1139  //remove space
1140  rempaceSpaces(part);
1141  //add number of removed spaces to ed
1142  topEd -= strlen(part);
1143  vector<std::unique_ptr<symspell::SuggestItem>> results;
1144  Lookup(part, symspell::Verbosity::Top, maxEditDistance, results);
1145  if (results.size() > 0)
1146  {
1147  size_t termLen = strlen(results[0]->term);
1148  topResult = new char[termLen + 1];
1149  std::memcpy(topResult, results[0]->term, termLen);
1150  topResult[termLen] = '\0';
1151 
1152  topEd += results[0]->distance;
1153  //Naive Bayes Rule
1154  //we assume the word probabilities of two words to be independent
1155  //therefore the resulting probability of the word combination is the product of the two word probabilities
1156 
1157  //instead of computing the product of probabilities we are computing the sum of the logarithm of probabilities
1158  //because the probabilities of words are about 10^-10, the product of many such small numbers could exceed (underflow) the floating number range and become zero
1159  //log(ab)=log(a)+log(b)
1160  topProbabilityLog = (double)log10((double)results[0]->count / (double)N);
1161  }
1162  else
1163  {
1164  delete[] topResult;
1165  topResult = part;
1166  //default, if word not found
1167  //otherwise long input text would win as long unknown word (with ed=edmax+1 ), although there there should many spaces inserted
1168  topEd += strlen(part);
1169  topProbabilityLog = (double)log10(10.0 / (N * pow(10.0, strlen(part))));
1170  }
1171 
1172  int destinationIndex = ((i + circularIndex) % arraySize);
1173 
1174  //set values in first loop
1175  if (j == 0)
1176  {
1177  compositions[destinationIndex]->set(_strdup(part), _strdup(topResult), topEd, topProbabilityLog);
1178  }
1179  else if ((i == maxSegmentationWordLength)
1180  //replace values if better probabilityLogSum, if same edit distance OR one space difference
1181  || (((compositions[circularIndex]->distanceSum + topEd == compositions[destinationIndex]->distanceSum) || (compositions[circularIndex]->distanceSum + separatorLength + topEd == compositions[destinationIndex]->distanceSum)) && (compositions[destinationIndex]->probabilityLogSum < compositions[circularIndex]->probabilityLogSum + topProbabilityLog))
1182  //replace values if smaller edit distance
1183  || (compositions[circularIndex]->distanceSum + separatorLength + topEd < compositions[destinationIndex]->distanceSum))
1184  {
1185  const char* segmented = compositions[circularIndex]->segmentedString;
1186  const char* corrected = compositions[circularIndex]->correctedString;
1187 
1188  size_t segmentedLen = strlen(segmented);
1189  size_t correctedLen = strlen(corrected);
1190  size_t partLen = strlen(part);
1191  size_t topResultLen = strlen(topResult);
1192 
1193  char* segmentedTmp = new char[segmentedLen + partLen + 2];
1194  char* correctedTmp = new char[correctedLen + topResultLen + 2];
1195 
1196  std::memcpy(segmentedTmp, segmented, segmentedLen);
1197  std::memcpy(segmentedTmp + segmentedLen, " ", 1);
1198  std::memcpy(segmentedTmp + segmentedLen + 1, part, partLen);
1199  segmentedTmp[segmentedLen + partLen + 1] = '\0';
1200 
1201  std::memcpy(correctedTmp, corrected, correctedLen);
1202  std::memcpy(correctedTmp + correctedLen, " ", 1);
1203  std::memcpy(correctedTmp + correctedLen + 1, topResult, topResultLen);
1204  correctedTmp[correctedLen + topResultLen + 1] = '\0';
1205 
1206  if (strlen(compositions[destinationIndex]->segmentedString) > 0)
1207  delete[] compositions[destinationIndex]->segmentedString;
1208 
1209  if (strlen(compositions[destinationIndex]->correctedString) > 0)
1210  delete[] compositions[destinationIndex]->correctedString;
1211 
1212  compositions[destinationIndex]->set(segmentedTmp,
1213  correctedTmp,
1214  compositions[circularIndex]->distanceSum + separatorLength + topEd,
1215  compositions[circularIndex]->probabilityLogSum + topProbabilityLog);
1216  }
1217  }
1218  ++circularIndex; if (circularIndex == arraySize) circularIndex = 0;
1219  }
1220  return compositions[circularIndex];
1221  }
